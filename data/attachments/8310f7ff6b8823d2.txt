// Set response object as internal variable
let jsonData = {};
try {jsonData = pm.response.json();}catch(e){}

// Set property value as variable
const _resToken = jsonData?.token;

// pm.collectionVariables - Set authToken as variable for jsonData.token
if (_resToken !== undefined) {
   pm.collectionVariables.set("authToken", jsonData.token);
   console.log("- use {{authToken}} as collection variable for value",jsonData.token);
} else {
   console.log('INFO - Unable to assign variable {{authToken}}, as jsonData.token is undefined.');
};
